100 decltype(auto) kleisli_compose(TLeft&& lhs, TRight&& rhs) {
101  using FirstArg = typename function::Info<TLeft>::template ArgType<0>;
102  using RhsReturnType = typename function::Info<TRight>::ReturnType;
103  static_assert(type::DomainTypeInfo<RhsReturnType>::hasMonadicBase, "Right hand side needs to have Monadic base");
104 
105  return [rhs = std::forward<TRight>(rhs), lhs = std::forward<TLeft>(lhs)](const FirstArg& arg) -> RhsReturnType {
106  using LhsReturnType = typename function::Info<TLeft>::ReturnType;
107  const LhsReturnType intermediate_result = lhs(arg);
108 
109  if constexpr (std::is_void_v<typename type::DomainTypeInfo<LhsReturnType>::ValueType>) {
110  if (!intermediate_result) {
111  return type::DomainTypeInfo<RhsReturnType>::handleError(intermediate_result);
112  } else {
113  return rhs();
114  }
115  } else {
116  if constexpr (type::DomainTypeInfo<typename function::Info<TRight>::template ArgType<0>>::hasMonadicBase) {
117  return rhs(intermediate_result);
118  } else {
119  if (!intermediate_result) {
120  return type::DomainTypeInfo<RhsReturnType>::handleError(intermediate_result);
121  }
122  return rhs(intermediate_result.value());
123  }
